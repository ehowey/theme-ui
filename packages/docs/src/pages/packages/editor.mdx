import merge from 'lodash.merge'

export const reducer = (state, next) => merge({}, state, next)

export const State = props => {
  const [value, setValue] = useReducer(reducer, props.value)
  return props.render([value, setValue])
}

# @theme-ui/editor

Components for editing Theme UI theme objects.

```sh
npm i @theme-ui/editor
```

```js
import {
  Editor,
  ColorPalette,
  Fonts,
  FontWeights,
  LineHeights,
  FontSizes,
  Space,
  ColorMode,
} from '@theme-ui/editor'
```
